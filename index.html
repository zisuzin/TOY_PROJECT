const clock = document.querySelector("#clock"); const deg = 6; const hr = document.querySelector("#hr"); const mn = document.querySelector("#mn"); const sc = document.querySelector("#sc"); function getAnalogClock() { const today = new Date(); const hh = today.getHours() * 30; const mm = today.getMinutes() * deg; const ss = today.getSeconds() * deg; hr.style.transform = `rotateZ(${hh + mm / 12}deg)`; mn.style.transform = `rotateZ(${mm}deg)`; sc.style.transform = `rotateZ(${ss}deg)`; } getAnalogClock(); setInterval (getAnalogClock, 1000);
